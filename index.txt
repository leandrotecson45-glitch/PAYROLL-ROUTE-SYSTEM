/* ================= GEO TAG ================= */
document.getElementById("photos").addEventListener("change",e=>{

  [...e.target.files].forEach(file=>{

    let r=new FileReader();

    r.onload=ev=>{

      let img=new Image();

      img.onload=()=>{

        EXIF.getData(img,function(){

          let lat=EXIF.getTag(this,"GPSLatitude");
          let lon=EXIF.getTag(this,"GPSLongitude");
          let latRef=EXIF.getTag(this,"GPSLatitudeRef");
          let lonRef=EXIF.getTag(this,"GPSLongitudeRef");

          if(lat && lon){

            let la=toDD(lat,latRef).toFixed(6);
            let lo=toDD(lon,lonRef).toFixed(6);

            addManual(la,lo);

          }
        });
      };

      img.src=ev.target.result;
    };

    r.readAsDataURL(file);
  });
});

/* ================= GPS CONVERT ================= */
function toDD(v,r){
  let d=v[0]+v[1]/60+v[2]/3600;
  return (r=="S"||r=="W")?-d:d;
}

/* ================= GET POINTS ================= */
function getPoints(){

  let pts=[];

  document.querySelectorAll(".card").forEach(c=>{

    let i=c.querySelectorAll("input");
    if(i.length<2) return;

    let a=parseFloat(i[0].value);
    let b=parseFloat(i[1].value);

    if(!isNaN(a)&&!isNaN(b)){
      pts.push([a,b]);
    }
  });

  return pts;
}

/* ================= ROUTE ================= */
async function generate(){

  let pts=getPoints();

  if(pts.length<2){
    alert("Need at least 2 points");
    return;
  }

  let coords=pts.map(p=>p[1]+","+p[0]).join(";");

  let url="https://router.project-osrm.org/route/v1/driving/"+coords+"?overview=full&geometries=geojson";

  try{

    let res=await fetch(url);
    let data=await res.json();

    if(!data.routes || !data.routes[0]){
      alert("No route found");
      return;
    }

    let route=data.routes[0].geometry.coordinates;

    let latlngs=route.map(p=>[p[1],p[0]]);

    if(line) map.removeLayer(line);

    line=L.polyline(latlngs,{
      color:"#0d6efd",
      weight:5
    }).addTo(map);

    map.fitBounds(line.getBounds());

    let km=data.routes[0].distance/1000;
    let rate=parseFloat(document.getElementById("rate").value||0);

    document.getElementById("result").innerHTML=
      "📏 TOTAL KM: "+km.toFixed(2)+"<br>"+
      "💰 TOTAL PAY: ₱"+(km*rate).toFixed(2);

  }catch(e){
    alert("Routing failed");
  }
}
